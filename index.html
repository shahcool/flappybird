const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const bird = { x: 50, y: 300, velocity: 0, radius: 20 }; const pipes = []; let score = 0; function drawBird() { ctx.beginPath(); ctx.arc(bird.x, bird.y, bird.radius, 0, Math.PI * 2); ctx.fillStyle = 'yellow'; ctx.fill(); ctx.closePath(); } function drawPipes() { for (let pipe of pipes) { ctx.fillStyle = 'green'; ctx.fillRect(pipe.x, 0, pipe.width, pipe.top); ctx.fillRect(pipe.x, pipe.bottom, pipe.width, canvas.height - pipe.bottom); } } function drawScore() { ctx.fillStyle = 'black'; ctx.font = '24px Arial'; ctx.fillText(`Score: ${score}`, 10, 30); } function update() { bird.velocity += 0.5; bird.y += bird.velocity; if (pipes.length === 0 || pipes[pipes.length - 1].x < canvas.width - 200) { const gap = 200; const pipeTop = Math.random() * (canvas.height - gap); pipes.push({ x: canvas.width, top: pipeTop, bottom: pipeTop + gap, width: 50, counted: false }); } for (let pipe of pipes) { pipe.x -= 2; if (pipe.x + pipe.width < bird.x && !pipe.counted) { score++; pipe.counted = true; } if (bird.x + bird.radius > pipe.x && bird.x - bird.radius < pipe.x + pipe.width && (bird.y - bird.radius < pipe.top || bird.y + bird.radius > pipe.bottom)) { gameOver(); } } pipes = pipes.filter(pipe => pipe.x + pipe.width > 0); if (bird.y + bird.radius > canvas.height || bird.y - bird.radius < 0) { gameOver(); } } function gameOver() { alert(`Game Over! Your score: ${score}`); bird.y = 300; bird.velocity = 0; pipes.length = 0; score = 0; } function gameLoop() { ctx.clearRect(0, 0, canvas.width, canvas.height); update(); drawBird(); drawPipes(); drawScore(); requestAnimationFrame(gameLoop); } canvas.addEventListener('click', () => { bird.velocity = -10; }); gameLoop();
